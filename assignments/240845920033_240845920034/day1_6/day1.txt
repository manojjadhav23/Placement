#include<iostream>
using namespace std;
int main() {
	int a, d, sum = 0;
	cout << "enter the number";
	cin >> a;
	int cp = a;
	for (; a != 0;) {
		d = a % 10;
		sum = sum + (d * d * d);
		a = a / 10;
	}
	if (sum == cp) {
		cout << "armstrong";
	}
	else
		cout << "not";
}




#include<iostream>
using namespace std;
int main() {
	int a,digit,rev=0;
	cout << "enter the number";
	cin >> a;
	int cp = a;
	for (; a != 0;) {
		digit = a % 10;
		rev= (rev*10)+digit;
		a = a / 10;
	}
	if (rev == cp) {
		cout << "palindrome";
	}
	else
		cout << "not a palindrome";
}

//Class structure with consructor
#include <iostream>  
using namespace std; 
class ComplexNumber
{

	int rl,img;
	public:
		void accept();
		ComplexNumber();
		ComplexNumber(int,int);
		
};
void ComplexNumber ::accept()
{
	cout<<"complex no is "<<rl<<"+"<<img<<"i"<<endl;
}
ComplexNumber::ComplexNumber()
{
	cout<<"in default constructor\n";

	cout<<"enter value of real\n";
	cin>>rl;
	cout<<"enter value of img\n";
	cin>>img;
	
}
ComplexNumber::ComplexNumber(int r,int i)
{
	cout<<"in parametrized constructor\n";
	rl=r;
	img=i;
}
int main()
{
	ComplexNumber d1;
	d1.accept();

	ComplexNumber d2(5,6);
	d2.accept();
	
}
